# install django
```
 pip3 install django
```

# create project
```
 django-admin startproject demo
 ```

# change */settings.py  ALLOWED_HOSTS = [*] to allow asscess

# start project
```
  python3 manage.py startapp demoapp
```

# run project
```
python manage.py runserver 8030
http://127.0.0.1:8030/
```
